docker run -it --name karting --hostname karting -v /home/lorak/Templates/docker_volumes/karting:/home/karting -w /home/karting debian /bin/bash

# Wszystko co poniżej już w konenerze
# w sumie to mozecie wszystko skopiować na raz i wkleić i sru enter
# tylko usuńcie te 2 komentarze po '#'

apt update && apt upgrade -y;

apt install -y git python-is-python3 python3.13-venv pkg-config default-libmysqlclient-dev build-essential python3.13-dev nodejs npm vim nano mariadb-server;

service mariadb start;

# to wklejacie całe aż do EOF
mysql -u root <<EOF
CREATE DATABASE IF NOT EXISTS karting;
EOF

# to tak samo całe
mysql -u root <<EOF
ALTER USER 'root'@'localhost' IDENTIFIED BY '';
FLUSH PRIVILEGES;
EOF


git clone https://github.com/atwardzik/Karting-Manager.git;

cd Karting-Manager;

chmod +x scripts/setup.sh

./scripts/setup.sh;

source .venv/bin/activate;


flask --app karting init-db;

flask --app karting run --debug --host=0.0.0.0;
